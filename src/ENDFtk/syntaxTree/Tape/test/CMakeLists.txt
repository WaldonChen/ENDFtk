add_executable( ENDFtk.syntaxTree.Tape.test Tape.test.cpp )
set( test_flags ${ENDFtk_compiler_flags} )
separate_arguments( test_flags_list UNIX_COMMAND "${test_flags}" )
foreach( flag IN LISTS test_flags_list )
    target_compile_options( ENDFtk.syntaxTree.Tape.test PUBLIC ${flag} )
endforeach( flag )
set_target_properties( ENDFtk.syntaxTree.Tape.test PROPERTIES LINK_FLAGS "${test_flags}" )
target_link_libraries( ENDFtk.syntaxTree.Tape.test PUBLIC ENDFtk )
file( GLOB resources "resources/*" ) 
file( COPY "${resources}" DESTINATION "${CMAKE_CURRENT_BINARY_DIR}" ) 
add_test( NAME ENDFtk.syntaxTree.Tape COMMAND ENDFtk.syntaxTree.Tape.test )
