add_executable( Section.test Section.test.cpp )
set( test_flags ${ENDFtk_compiler_flags} )
separate_arguments( test_flags_list UNIX_COMMAND "${test_flags}" )
foreach( flag IN LISTS test_flags_list )
    target_compile_options( Section.test PUBLIC ${flag} )
endforeach( flag )
set_target_properties( Section.test PROPERTIES LINK_FLAGS "${test_flags}" )
target_link_libraries( Section.test PUBLIC ENDFtk )
add_test( NAME Section COMMAND Section.test )
