add_executable( ENDFtk.section.Base.test Base.test.cpp )
set( test_flags ${ENDFtk_compiler_flags} )
separate_arguments( test_flags_list UNIX_COMMAND "${test_flags}" )
foreach( flag IN LISTS test_flags_list )
    target_compile_options( ENDFtk.section.Base.test PUBLIC ${flag} )
endforeach( flag )
set_target_properties( ENDFtk.section.Base.test PROPERTIES LINK_FLAGS "${test_flags}" )
target_link_libraries( ENDFtk.section.Base.test PUBLIC ENDFtk )
add_test( NAME ENDFtk.section.Base COMMAND ENDFtk.section.Base.test )
