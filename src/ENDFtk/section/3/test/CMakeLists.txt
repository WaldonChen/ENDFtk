add_executable( 3.test 3.test.cpp )
set( test_flags ${ENDFtk_compiler_flags} )
separate_arguments( test_flags_list UNIX_COMMAND "${test_flags}" )
foreach( flag IN LISTS test_flags_list )
    target_compile_options( 3.test PUBLIC ${flag} )
endforeach( flag )
set_target_properties( 3.test PROPERTIES LINK_FLAGS "${test_flags}" )
target_link_libraries( 3.test PUBLIC ENDFtk )
add_test( NAME 3 COMMAND 3.test )
