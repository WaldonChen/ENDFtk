add_executable( ENDFtk.file.Type.test Type.test.cpp )
set( test_flags ${ENDFtk_compiler_flags} )
separate_arguments( test_flags_list UNIX_COMMAND "${test_flags}" )
foreach( flag IN LISTS test_flags_list )
    target_compile_options( ENDFtk.file.Type.test PUBLIC ${flag} )
endforeach( flag )
set_target_properties( ENDFtk.file.Type.test PROPERTIES LINK_FLAGS "${test_flags}" )
target_link_libraries( ENDFtk.file.Type.test PUBLIC ENDFtk )
file( GLOB resources "resources/*" ) 
file( COPY "${resources}" DESTINATION "${CMAKE_CURRENT_BINARY_DIR}" ) 
add_test( NAME ENDFtk.file.Type COMMAND ENDFtk.file.Type.test )
